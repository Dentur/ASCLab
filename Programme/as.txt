{	
	set_marker THIS FORWARD 0
	while((position_x != goal_x) AND (position_y != goal_y))
	{
		//Select lowest maker
		int index;
		index = 0;
		int minIndex;
		int minVal;
		minIndex = -1;
		minVal = 99999;
		while(index < get_marker_count)
		{
			int curVal; 
			curVal = get_marker index;
			if(curVal < minVal)
			{
				minVal = curVal;
				minIndex = index;
			}
		}
		if(minIndex == -1)
		{
			print "Error: Goal cannot be reached (1)";
			break;
		}
		//Teleport to marker
		teleport minIndex;
		//Set surrounding markers
		if(look NORTH == FREE)
		{
			set_marker NORTH SOUTH curVal+1
		}
		if(look NORTH == MARKER)
		{
			if(marker_value NORTH >= 0)
			{
				if((curVal+1) < marker_value NORTH)
				{
					set_marker NORTH SOUTH curVal+1
				}
			}
		}
		if(look EAST == FREE)
		{
			set_marker EAST WEST curVal+1
		}
		if(look EAST == MARKER)
		{
			if(marker_value EAST >= 0)
			{
				if((curVal+1) < marker_value EAST)
				{
					set_marker EAST WEST curVal+1
				}
			}
		}
		if(look SOUTH == FREE)
		{
			set_marker SOUTH NORTH curVal+1
		}
		if(look SOUTH == MARKER)
		{
			if(marker_value SOUTH >= 0)
			{
				if((curVal+1) < marker_value SOUTH)
				{
					set_marker SOUTH NORTH curVal+1
				}
			}
		}
		if(look WEST == FREE)
		{
			set_marker WEST EAST curVal+1
		}
		if(look WEST == MARKER)
		{
			if(marker_value WEST >= 0)
			{
				if((curVal+1) < marker_value WEST)
				{
					set_marker WEST EAST curVal+1
				}
			}
		}
		//Remove this marker
		delete_marker minIndex;
	}
	if(position_x == goal_x)
	{
		print "Ziel gefunden";
	}
}
